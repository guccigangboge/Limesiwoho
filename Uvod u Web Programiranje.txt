                    prost = false;

            printf("%d\n", i);
            n--;
        }
    }

    return 0;
}

bool jeProst(int n) {
    if (n == 1 || n == 0) return false;
    if (n == 2) return true;
    
    for (int i = 2; i <= sqrt(n); i++) {
        if (n % i == 0)
            return false;
    }
    return true;
}

int main() {
    int n, br = 1;
    scanf_s("%d", &n);

    while (true) {
        if (jeProst(n + br)) {
            printf("%d", n + br);
            break;
        }
        br = -br;
        if (br > 0)
            br++;
    }

    return 0;
}

#include <math.h>

float doubleFactoriel(float n) {
    if (n == 1 || n == 0)
        return 1;
    return n * doubleFactoriel(n - 2);
}

int sumaOnogSranja(int n) {
    int sum = 0;
    while (true) {
        int digit = n % 10;
        sum += pow(doubleFactoriel(digit), 2);

        n /= 10;

        if (n == 0)
            return sum;
    }
}

int main() {

    int n;
    scanf_s("%d", &n);

    for (int i = 0; i < n; i++) {
        if (sumaOnogSranja(i) == i)
            printf("%d \n", i);
    }

    return 0;
}

void ispis(int realni, int imaginarni) {
    if (imaginarni > 0)
        if (realni != 0)
            printf("%d + %di\n", realni, imaginarni);
        else
            printf("%di\n", imaginarni);
    if (imaginarni < 0)
        if (realni != 0)
            printf("%d - %di\n", realni, abs(imaginarni));
        else
            printf("%di\n", imaginarni);
    if (imaginarni == 0)
        printf("%d\n", realni);
}

void ispis_f(float realni, float imaginarni) {
    if (imaginarni > 0)
        if (realni != 0)
            printf("%f + %fi\n", realni, imaginarni);
        else
            printf("%fi\n", imaginarni);
    if (imaginarni < 0)
        if (realni != 0)
            printf("%f - %fi\n", realni, fabs(imaginarni));
        else
            printf("%fi\n", imaginarni);
    if (imaginarni == 0)
        printf("%f\n", realni);
}

void zbir(int realni1, int imaginarni1, int realni2, int imaginarni2, int& zbirR, int& zbirI) {
    zbirR = realni1 + realni2;
    zbirI = imaginarni1 + imaginarni2;
}

void razlika(int realni1, int imaginarni1, int realni2, int imaginarni2, int& razlikaR, int& razlikaI) {
    razlikaR = realni1 - realni2;
    razlikaI = imaginarni1 - imaginarni2;
}

void proizvod(int realni1, int imaginarni1, int realni2, int imaginarni2, int& proizvodR, int& proizvodI) {
    proizvodR = (realni1 * realni2) - (imaginarni1 * imaginarni2);
    proizvodI = (realni1 * realni2) + (imaginarni1 * imaginarni2);
}

void kolicnik(int realni1, int imaginarni1, int realni2, int imaginarni2, float& kolicnikR, float& kolicnikI) {
    kolicnikR = (((realni1 * realni2) + (imaginarni1 * imaginarni2)) / (pow(realni2, 2) + pow(imaginarni2, 2)));
    kolicnikI = (((realni1 * realni2) - (imaginarni1 * imaginarni2)) / (pow(realni2, 2) + pow(imaginarni2, 2)));
}
